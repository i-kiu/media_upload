document.addEventListener('DOMContentLoaded', function () { $('.parsley-required:not(.parsley-required-hide)').each(function() { validationMessage = $(this).attr('data-validation-message'); $(this).find('input[type=hidden]').attr({ 'required': true, 'data-parsley-require': true, 'data-parsley-trigger': 'keyup', 'data-parsley-required-message': validationMessage }); }); $('form').each(function(){ $(this).parsley({ excluded: 'input[type=button], input[type=submit], input[type=reset]', inputs: 'input, textarea, select, input[type=hidden], :hidden', triggerAfterFailure: 'focusout changed.bs.select', trigger: 'change', successClass: "has-success", errorClass: "has-error", classHandler: function (el) { return el.$element.closest('.form-group'); } }); }); });
